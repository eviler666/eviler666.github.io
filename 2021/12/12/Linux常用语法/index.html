linux运维

mkdir #创建目录（mkdir -p BBB/Test）
chmod -R 777 目录 #授权
chown -R root：root 目录 #更改文件夹拥有者
chgrp  用户组名 文件名		#x

ps –ef| grep 进程名	#查进程 例如：ps -ef|grep tomcat
kill -9 进程id	#杀进程 例如： kill -9 72666
ps –ef| grep 进程名	#在查 例如：ps -ef|grep tomcat
./启动程序.sh -d	#重启 例如：./startup.sh
ps –ef| grep 进程名	#在查 例如：ps -ef|grep tomcat
